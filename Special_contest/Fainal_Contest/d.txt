
#include <bits/stdc++.h>
using namespace std;

static const long long MOD = 1000000007;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<long long> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    vector<int> prev(n), next(n);
    unordered_map<long long, int> lastSeen;

    // Compute prev[]
    for (int i = 0; i < n; i++) {
        if (lastSeen.count(a[i])) {
            prev[i] = lastSeen[a[i]];
        } else {
            prev[i] = -1;
        }
        lastSeen[a[i]] = i;
    }

    lastSeen.clear();

    // Compute next[]
    for (int i = n - 1; i >= 0; i--) {
        if (lastSeen.count(a[i])) {
            next[i] = lastSeen[a[i]];
        } else {
            next[i] = n;
        }
        lastSeen[a[i]] = i;
    }

    long long ans = 0;

    for (int i = 0; i < n; i++) {
        long long left = i - prev[i];
        long long right = next[i] - i;
        long long contrib = (a[i] % MOD) * (left % MOD) % MOD * (right % MOD) % MOD;
        ans = (ans + contrib) % MOD;
    }

    cout << ans << "\n";
    return 0;
}
